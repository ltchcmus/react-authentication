\section{TỔNG QUAN DỰ ÁN}

\subsection{Giới thiệu}

UserHub là một hệ thống xác thực người dùng hoàn chỉnh (Full Stack Authentication System) được xây dựng theo yêu cầu của bài tập IA06. Dự án triển khai một ứng dụng web hiện đại với backend API sử dụng NestJS và frontend React, cung cấp chức năng đăng ký và đăng nhập người dùng với giao diện thân thiện và bảo mật.

\subsection{Mục tiêu dự án}

\begin{itemize}[leftmargin=*]
    \item Xây dựng backend API với NestJS để xử lý đăng ký và quản lý dữ liệu người dùng
    \item Phát triển frontend React với giao diện đẹp mắt, responsive
    \item Tích hợp hoàn chỉnh giữa frontend và backend
    \item Triển khai ứng dụng lên môi trường production
    \item Đảm bảo bảo mật và trải nghiệm người dùng tốt nhất
\end{itemize}

\subsection{Chi tiết hoàn thành các yêu cầu}

\subsubsection{Backend Implementation (4/4 điểm)}

\textbf{1. Database Setup} (\textcolor{successcolor}{✓ Hoàn thành})

\begin{itemize}
    \item User schema với các trường: \texttt{userId} (UUID), \texttt{email} (unique), \texttt{password} (hashed), \texttt{createdAt}
    \item Sử dụng TypeORM entities với decorators để định nghĩa schema
    \item Unique constraint cho email để tránh duplicate
    \item Auto-generate UUID và timestamp
\end{itemize}

\textbf{2. API Endpoint - POST /user/register (2/2 điểm)} (\textcolor{successcolor}{✓ Hoàn thành})

\begin{itemize}
    \item Endpoint: \texttt{POST /api/v1/users/register}
    \item Validate input với class-validator (email format, password length 6-50 ký tự)
    \item Check duplicate email trước khi tạo user mới
    \item Hash password với bcrypt (10 salt rounds)
    \item Return HttpResponse wrapper với status code phù hợp
    \item Response bao gồm: userId, email, createdAt (không trả về password)
\end{itemize}

\textbf{3. Error Handling (2/2 điểm)} (\textcolor{successcolor}{✓ Hoàn thành})

\begin{itemize}
    \item Custom AppException class với error codes
    \item Global Exception Filter để xử lý tất cả errors
    \item Validation errors với thông báo rõ ràng (field-level errors)
    \item Database errors được catch và format thành thông báo dễ hiểu
    \item HTTP status codes chuẩn (200, 400, 404, 409, 500)
\end{itemize}

\textbf{4. Security} (\textcolor{successcolor}{✓ Hoàn thành})

\begin{itemize}
    \item Environment variables (.env) cho database URI, API keys
    \item CORS được enable cho frontend URL
    \item Password hashing với bcrypt
    \item Input sanitization và validation
    \item API Key middleware (optional layer)
\end{itemize}

\end{itemize}

\textbf{2. Form Validation} (\textcolor{successcolor}{✓ Hoàn thành})

\textbf{3. Routing (1/1 điểm)} (\textcolor{successcolor}{✓ Hoàn thành})

\section{FUNCTIONAL VÀ NON-FUNCTIONAL REQUIREMENTS}
%------------------------------------------------------------

\subsection{Functional Requirements}

\subsubsection{Đã triển khai theo yêu cầu}

\begin{enumerate}
    \item \textbf{User Registration}
    \begin{itemize}
        \item User có thể đăng ký tài khoản mới
        \item Validate email format và password length
        \item Check duplicate email
        \item Hash password trước khi lưu
        \item Hiển thị success/error messages
    \end{itemize}
    
    \item \textbf{User Login}
    \begin{itemize}
        \item User có thể đăng nhập với email/password
        \item Verify password với bcrypt compare
        \item Tạo session sau khi login thành công
        \item Redirect đến dashboard
    \end{itemize}
    
    \item \textbf{Navigation}
    \begin{itemize}
        \item Điều hướng giữa Home, Login, Sign Up, Dashboard
        \item Protected routes cho authenticated pages
        \item Auto redirect khi chưa login
    \end{itemize}
    
    \item \textbf{User Profile}
    \begin{itemize}
        \item Hiển thị user information
        \item Logout functionality
        \item Account statistics
    \end{itemize}
\end{enumerate}

\subsection{Non-Functional Requirements}

\subsubsection{Performance}

\begin{itemize}
    \item \textbf{Response Time:} API response < 500ms
    \item \textbf{Loading States:} Loading indicators cho tất cả async operations
    \item \textbf{Caching:} React Query cache để giảm API calls
    \item \textbf{Build Optimization:} Vite build với code splitting
\end{itemize}

\subsubsection{Security}

\begin{itemize}
    \item \textbf{Password Hashing:} bcrypt với 10 salt rounds
    \item \textbf{Input Validation:} class-validator trên backend, React Hook Form trên frontend
    \item \textbf{CORS:} Chỉ allow requests từ frontend domain
    \item \textbf{Environment Variables:} Sensitive data không hard-code
    \item \textbf{HTTPS:} Bắt buộc trên production
\end{itemize}

\subsubsection{Usability}

\begin{itemize}
    \item \textbf{Responsive Design:} Hoạt động tốt trên mobile/tablet/desktop
    \item \textbf{Clear Feedback:} Success/error messages rõ ràng
    \item \textbf{Accessibility:} ARIA labels, keyboard navigation
    \item \textbf{Visual Design:} Professional gradient theme, consistent styling
\end{itemize}

\subsubsection{Maintainability}

\begin{itemize}
    \item \textbf{Code Organization:} Clean architecture, separation of concerns
    \item \textbf{TypeScript:} Type safety trên backend
    \item \textbf{Documentation:} Comprehensive README và guides
    \item \textbf{Error Logging:} Winston logger cho debugging
\end{itemize}

\subsubsection{Scalability}

\begin{itemize}
    \item \textbf{Database:} PostgreSQL với indexing
    \item \textbf{Modular Architecture:} NestJS modules dễ extend
    \item \textbf{API Design:} RESTful endpoints
    \item \textbf{Deployment:} Serverless-ready (Vercel, Render)
\end{itemize}

\subsection{B. API Endpoints Documentation}

\textbf{Base URL:} \texttt{http://localhost:9999/api/v1}

\subsubsection{POST /users/register}

\textbf{Request Body:}
\begin{verbatim}
{
  "email": "user@example.com",
  "password": "password123",
  "confirmPassword": "password123"
}
\end{verbatim}

\textbf{Success Response (200):}
\begin{verbatim}
{
  "code": 200,
  "message": "User registered successfully",
  "data": {
    "userId": "uuid-string",
    "email": "user@example.com",
    "createdAt": "2025-12-01T10:30:00.000Z"
  }
}
\end{verbatim}

\textbf{Error Response (400):}
\begin{verbatim}
{
  "code": 400,
  "message": "Validation failed",
  "data": [
    {
      "field": "email",
      "constraints": ["Invalid email format"]
    }
  ]
}
\end{verbatim}

\textbf{Request Body:}
